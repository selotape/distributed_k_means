# download and extract datasets
### COVTYPE
cd data_samples/covtype && wget https://archive.ics.uci.edu/ml/machine-learning-databases/covtype/covtype.data.gz && gunzip covtype.data.gz && cd -
### KDDCUP99
cd data_samples/kddcup99 && wget http://kdd.ics.uci.edu/databases/kddcup99/kddcup.data.gz && gunzip kddcup.data.gz  && cd -
### SKIN
mkdir -p data_samples/skin && cd data_samples/skin  &&  wget https://archive.ics.uci.edu/ml/machine-learning-databases/00229/Skin_NonSkin.txt && cd -
### BIGCROSS
mkdir -p data_samples/bigcross && cd data_samples/bigcross  &&  wget https://s3.amazonaws.com/h2o-training/clustering/BigCross.data.gz && gunzip BigCross.data.gz && cd -
### POWER CONSUMPTION
mkdir -p data_samples/power && cd data_samples/power  &&  wget https://archive.ics.uci.edu/ml/machine-learning-databases/00235/household_power_consumption.zip && unzip household_power_consumption.zip && cd -
### Activity Recognition
mkdir -p data_samples/activity && cd data_samples/activity  &&  wget https://archive.ics.uci.edu/ml/machine-learning-databases/00344/Activity%20recognition%20exp.zip && mv Activity\ recognition\ exp.zip Activity_recognition_exp.zip && unzip Activity_recognition_exp.zip && cd -
### HIGGS
mkdir -p data_samples/higgs && cd data_samples/higgs  &&  wget https://archive.ics.uci.edu/ml/machine-learning-databases/00280/HIGGS.csv.gz && gunzip HIGGS.csv.gz && cd -
